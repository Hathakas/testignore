using INab.Dissolve;
using UnityEngine;
using UnityEngine.Video;

public class GameSphere : MonoBehaviour
{
   public Material[] materials; // 0: Normal, 1: Inverted
   public VideoPlayer videoPlayer;
   public Renderer sphereRenderer;
   public SphereContent activeContent;
   public SphereContent inactiveContent;
   public Dissolver myDissolver;

   public int activeMaterialIndex; // 0 for normal, 1 for inverted
}
[System.Serializable]
public class SphereContent
{
   public ContentType contentType;
   public VideoClip videoClip;
   public Texture image;
   public SphereContent invertedContent; // Reference to the inverted world version
}
