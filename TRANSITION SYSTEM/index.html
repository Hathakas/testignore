4. TransitionManager.cs
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.Video;

public enum ContentType
{
    Video,
    Image
}

public class TransitionManager : MonoBehaviour
{
    public List<GameSphere> gameSpheres;
    public Texture videoBaseTexture; // Common texture used for video rendering
    public GameSphere currentSphere;

    private void Start()
    {
        // Initialize current sphere and content states
        currentSphere = gameSpheres[0];
    }

    public void ToggleInvertedWorld()
    {
        // Determine the inactive material index
        int inactiveMaterialIndex = (currentSphere.activeMaterialIndex == 0) ? 1 : 0;

        // Swap content to the inactive material before starting the transition
        SwapContent(inactiveMaterialIndex);

        if (inactiveMaterialIndex == 0)
        {
            // Material 0 is inactive, so we materialize it
            currentSphere.myDissolver.Materialize();
        }
        else
        {
            // Material 1 is inactive, so we dissolve it
            currentSphere.myDissolver.Dissolve();
        }

        // Update the active material index
        currentSphere.activeMaterialIndex = inactiveMaterialIndex;
    }

    public void SphereToSphereDissolve(GameSphere toSphere)
    {
        currentSphere = toSphere;
        ToggleInvertedWorld();
    }

    private void SwapContent(int inactiveMaterialIndex)
    {
        // Switch content on the inactive material
        SphereContent targetContent = (inactiveMaterialIndex == 0) ? currentSphere.activeContent : currentSphere.activeContent.invertedContent;
        
        if (targetContent.contentType == ContentType.Video)
        {
            // Set the video clip and play it
            currentSphere.videoPlayer.clip = targetContent.videoClip;
            currentSphere.videoPlayer.Play();

            // Assign the video base texture to the inactive material
            currentSphere.sphereRenderer.materials[inactiveMaterialIndex].mainTexture = videoBaseTexture;
        }
        else
        {
            // Assign the image texture to the inactive material
            currentSphere.sphereRenderer.materials[inactiveMaterialIndex].mainTexture = targetContent.image;
        }
    }
}
